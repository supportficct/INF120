    Public Function Vfibo() As Boolean
        Dim a, b, c As Integer
        a = -1 : b = 1
        Do
            c = a + b
            a = b : b = c
        Loop Until (c = n) Or (c > n)
        Return c = n
    End Function

    Public Function Fact(nu As Integer)
        Dim f As Double : Dim i As Integer

        If nu <> 0 Then
            f = 1
            For i = Math.Abs(nu) To 1 Step -1
                f = f * i
            Next
            If nu < 0 Then
                f = f * (-1)
            End If
        Else
            f = 1
        End If
        Return f
    End Function

    Public Function Vprim()
        Dim b As Boolean
        Dim c As Integer
        b = True : c = 0

        For i = 1 To n
            If (n Mod i) = 0 Then
                c = c + 1
            End If

            If c > 3 Then
                b = False
                i = n
            End If
        Next

        Return b

    End Function

        d = New Nent

        While n > 0
            d.n = n Mod 10 ^ nd
